1.AndroidManifest

<application …>
    <activity android:name=".MainActivity”>
        <intent-filter>//首先启动
            <action android:name="android.intent.action.MAIN"/>
            <category android:name=“android.intent.category.LAUNCHER"/>
    </activity>//注册
    
</application>

隐启动
<intent-filter>
    <action android:name=“ACTION_START” //内容不限，最好 包名+参数
    <category android:name=“android.intent.category.DEFAULT” //默认参数，此时启动不用加入addCategory
    <category android:name=“CATEGORY_START” // 自定义参数
</intent-filter>

startActivity(new Intent(“ACTION_START”).addCategory(“CATEGORY_START”))
